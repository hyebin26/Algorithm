## while을 사용한 풀이
```jsx
function solution(n, arr1, arr2) {
    var answer = [];
    let map1 = [];
    let map2 = [];
    function binary(num,leng){
        const arr = [];
        while(true){
            if(num % 2 === 0){
                arr.push(0);
            }else{
                arr.push(1);
            }
            num = Math.floor(num / 2);
            if(num === 1 || num === 0){
                arr.push(num); 
                //만약 arr length 가 num 보다 작으면 같아질 때 까지 0추가
                if(arr.length < leng){
                  for(let i= arr.length; i<leng ;i++){
                      arr.push(0);
                  }
                }
                break;
            }
        }
        return arr.reverse();
    }
    arr1.map((item) =>{
       map1.push(binary(item,n));
    });
    arr2.map((item)=>{
        map2.push(binary(item,n));
    });
    // 두 배열 중에 1이 포함되있으면 # 아니면 공백으로 추가하기?
    for(let i=0; i<n; i++){
        let st = "";
        for(let j=0; j<n; j++){
            if(map2[i][j] === 1 || map1[i][j] === 1){
                st += "#"; 
            }else{
                st += " "; 
            }
        }
        answer.push(st);
    }
    return answer;
};
```
